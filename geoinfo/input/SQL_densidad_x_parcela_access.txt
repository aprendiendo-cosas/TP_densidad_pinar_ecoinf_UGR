SELECT DICC_PARCELAS.ID_PARCELA, DICC_PARCELAS.UTM_X_GPS, DICC_PARCELAS.UTM_Y_GPS, PARCELA_DATOS_CEPAS_M_BAJO.DENSIDAD_TOTAL AS den_tot_bajo, PARCELA_DATOS_PIES_M_ALTO.DENSIDAD_TOTAL AS den_tot_alto, [den_tot_bajo]+[den_tot_alto] AS den_tot
FROM ((DICC_PARCELAS INNER JOIN VISITAS_PARCELAS ON DICC_PARCELAS.ID_PARCELA = VISITAS_PARCELAS.ID_PARCELA) INNER JOIN PARCELA_DATOS_CEPAS_M_BAJO ON VISITAS_PARCELAS.COD_VISITA_PARCELA = PARCELA_DATOS_CEPAS_M_BAJO.COD_VISITA_PARCELA) INNER JOIN PARCELA_DATOS_PIES_M_ALTO ON VISITAS_PARCELAS.COD_VISITA_PARCELA = PARCELA_DATOS_PIES_M_ALTO.COD_VISITA_PARCELA;
